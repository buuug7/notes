## 判断一个字符串中出现次数最多的字符，统计这个次数

```javascript
const str = "asdfssaaasasasasaa";
const result = {};

for (let i = 0; i < str.length; i++) {
  if (!result[str.charAt(i)]) {
    result[str.charAt(i)] = 1;
  } else {
    result[str.charAt(i)] += 1;
  }
}

let max = 0;

for (let item in result) {
  if (result[item] > max) {
    max = result[item];
  }
  console.log("item", item);
}

console.log(result);
console.log("max=", max);
```
